<body>

    <!-- Container for the uploader interface -->

    <div class="uploader">

        <h1>Upload MP3 File</h1> <!-- Header for the uploader -->

        <form id="uploadForm" enctype="multipart/form-data" onsubmit="event.preventDefault(); uploadFile();"> <!-- Form to handle file upload -->

            <input type="file" id="mp3File" name="mp3File" accept="audio/mpeg" required /> <!-- Input field for selecting MP3 file -->

            <button type="submit">Upload</button> <!-- Button to trigger file upload -->

            <p id="errorMessage" class="message error-message" style="display: none;"></p>

            <p id="successMessage" class="message success-message" style="display: none;"></p>

        </form>

    </div>


    <script>

        // Define the server URL dynamically for AWS Amplify or fallback to local

        const SERVER_URL = '/api'; // Assuming Amplify hosting with a configured API path


        // Function to handle the file upload process

        async function uploadFile() {

            const fileInput = document.getElementById('mp3File'); // Gets the file input element

            const errorMessage = document.getElementById('errorMessage'); // Gets the error message element

            const successMessage = document.getElementById('successMessage'); // Gets the success message element

            const formData = new FormData(); // Creates a FormData object to send file data


            // Checks if a file has been selected

            if (fileInput.files.length === 0) {

                errorMessage.style.display = 'block'; // Shows the error message

                errorMessage.textContent = 'Please select a file first.';

                successMessage.style.display = 'none'; // Hides success message

                return;

            } else {

                errorMessage.style.display = 'none'; // Hides the error message if the file is selected

            }


            formData.append('mp3File', fileInput.files[0]); // Appends the selected file to the FormData object


            try {

                // Sends the file to the server using the fetch API

                const response = await fetch(`${SERVER_URL}/upload`, {

                    method: 'POST', // Specifies the request method

                    body: formData // Includes the file data in the request body

                });


                // Handles the server response

                if (response.ok) {

                    const result = await response.json(); // Parses the JSON response

                    successMessage.style.display = 'block';

                    successMessage.textContent = `File uploaded successfully! Response: ${JSON.stringify(result)}`;

                } else {

                    const errorDetails = await response.text(); // Get server error details

                    errorMessage.style.display = 'block';

                    errorMessage.textContent = `Upload failed: ${errorDetails}`;

                    successMessage.style.display = 'none';

                }

            } catch (error) {

                // Logs and alerts in case of an error

                console.error('Error:', error);

                errorMessage.style.display = 'block';


                if (error.name === 'TypeError') {

                    errorMessage.textContent = 'Network error: Please check your connection or server availability.';

                } else if (error.message.includes('size')) {

                    errorMessage.textContent = 'File too large. Please upload a smaller file.';

                } else {

                    errorMessage.textContent = 'An unexpected error occurred. Please try again later.';

                }


                successMessage.style.display = 'none';

            }

        }

    </script>
